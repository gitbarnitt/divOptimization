debug_fit_gjam_model <- function(siteID, path = "data/plant_data.rds") {
  library(dplyr)
  library(tidyr)
  library(gjam)
  
  message("ğŸ“¦ Loading NEON data from: ", path)
  all_data <- readRDS(path)
  
  message("ğŸ” Subsetting site data for: ", siteID)
  site_data <- all_data %>% filter(siteID == !!siteID)
  
  if (nrow(site_data) == 0) {
    stop("âŒ No data found for siteID: ", siteID)
  }

  message("âœ… Data loaded. Proceeding with model prep...")

  # Check for sufficiency
  if (nrow(site_data) < 5 || length(unique(site_data$taxonID)) < 2) {
    message("âŒ Insufficient data to fit model.")
    return(NULL)
  }

  message("ğŸ”„ Pivoting to wide format...")
  y_data <- site_data %>%
    tidyr::pivot_wider(
      id_cols = c(siteID, year, plotID, nlcdClass),
      names_from = taxonID,
      values_from = mean_cover,
      values_fill = 0
    )

  message("ğŸ“Š Preparing covariates (x_data)...")
  x_data <- y_data %>%
    select(year, nlcdClass) %>%
    mutate(
      year = as.numeric(year),
      nlcdClass = as.factor(nlcdClass)
    )

  message("ğŸ“ˆ Preparing response matrix (y_matrix)...")
  y_matrix <- y_data %>%
    select(-siteID, -plotID, -year, -nlcdClass)

  if (ncol(y_matrix) < 2) {
    message("âŒ Too few species columns in y_matrix.")
    return(NULL)
  }

  message("ğŸ§¼ Filtering species columns with all zeros...")
  nonzero_cols <- y_matrix[, colSums(y_matrix, na.rm = TRUE) > 0, drop = FALSE]
  if (ncol(nonzero_cols) < 2) {
    message("âŒ Too few nonzero species.")
    return(NULL)
  }

  if (any(is.na(nonzero_cols))) {
    message("âŒ NA values found in y_matrix â€” aborting.")
    return(NULL)
  }

  if (!all(sapply(nonzero_cols, is.numeric))) {
    message("âŒ Non-numeric columns in y_matrix â€” aborting.")
    return(NULL)
  }

  formula <- ~ year + nlcdClass
  model_list <- list(
    formula = formula,
    xdata = x_data,
    ydata = nonzero_cols,
    modelList = list(typeNames = rep("Q", ncol(nonzero_cols)))
  )

  message("ğŸš€ Attempting to fit GJAM model...")
  fit <- tryCatch({
    gjam::gjam(
      formula = model_list$formula,
      xdata = model_list$xdata,
      ydata = model_list$ydata,
      modelList = model_list$modelList
    )
  }, error = function(e) {
    message("âŒ GJAM fit failed: ", e$message)
    return(NULL)
  })

  if (!is.null(fit)) {
    message("âœ… GJAM model fit succeeded.")
  }

  return(fit)
}
